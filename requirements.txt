# FAISS — installed via conda in the Dockerfile, NOT via pip.
#
# cuVS-enabled faiss-gpu is only published on conda channels (pytorch +
# nvidia + rapidsai), there's no PyPI wheel for it. The Dockerfile runs
# `conda install -c pytorch -c nvidia -c rapidsai -c conda-forge
# faiss-gpu-cuvs` so this requirements.txt deliberately omits faiss.
#
# Why cuVS matters: without it, GpuIndexIVFPQ is limited to M in
# {1,2,3,4,8,12,16,20,24,28,32,40,48,56,64,96}, so M=256 builds had to
# either fall to CPU (hours per multi-million-vector city) or get
# auto-shrunk to M=96 with a ~2% recall hit. cuVS removes the M cap
# entirely (any M <= FEATURE_DIM) and improves build/search speed.

# safetensors retained for any consumer of MegaLoc weights at index time;
# torch / torchvision / timm come from the pytorch base image.
safetensors>=0.3.0

# Polygon lookup for the region_polygon_simplified field in config.json.

# and resolves the city's admin polygon, so the inference server has it
# without needing to run the offline backfill.
shapely>=2.0.0
